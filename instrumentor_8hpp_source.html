46  void BeginSession(std::string const& name, std::string const& filepath = "results.json")
47  {
48  std::lock_guard<std::mutex> lock(mutex_);
49  if (currentSession_ != nullptr)
50  {
51  // If there is already a current session, then close it before beginning new one.
52  // Subsequent profiling output meant for the original session will end up in the
53  // newly opened session instead. That's better than having badly formatted
54  // profiling output.
55  if (Log::GetCoreLogger())
56  { // Edge case: BeginSession() might be before Log::Init()
57  MOCI_CORE_ERROR("Instrumentor::BeginSession('{0}') when session '{1}' already open.", name,
58  currentSession_->Name);
59  }
60  InternalEndSession();
61  }

63  buffer_.reserve(1'000'000);
64

66  if (outputStream_.is_open())

68  currentSession_ = MakeScope<InstrumentationSession>();

72  else
